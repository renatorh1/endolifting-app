<script>

  let bio='mixto';


  function setBio(v){

    bio=v;

    bMixto.classList.toggle('active', v==='mixto');

    bDelgado.classList.toggle('active', v==='delgado');

    bGraso.classList.toggle('active', v==='graso');


    const prioridad = document.getElementById('prioridad');

    const explica = document.getElementById('explica');


    if(v==='delgado'){

      prioridad.textContent='1470 dominante · 980 mínimo';

      explica.textContent='Objetivo: tensar sin vaciar. 980 en exceso envejece la cara.';

      pPeri.textContent='2–4';

      pMedio.textContent='3–4';

      pCuello.textContent='3–5';

    }else if(v==='graso'){

      prioridad.textContent='980 protagonista + 1470 soporte';

      explica.textContent='Objetivo: reducir volumen + retracción. 980 más presente en jowls/submentón/cuello.';

      pPeri.textContent='2–4';

      pMedio.textContent='4–7';

      pCuello.textContent='5–8';

    }else{

      prioridad.textContent='1470 base + 980 selectivo';

      explica.textContent='Mejilla: preservar volumen. 980 donde sobra grasa (jowls/submentón).';

      pPeri.textContent='2–4';

      pMedio.textContent='3–6';

      pCuello.textContent='4–6';

    }

  }


  function calc(){

    const W = parseFloat(document.getElementById('W').value||'0');

    const T = parseFloat(document.getElementById('T').value||'0');

    const A = parseFloat(document.getElementById('A').value||'1');

    const J = W*T;

    const F = J/A;

    outJ.textContent = isFinite(J) ? J.toFixed(0) : '—';

    outF.textContent = isFinite(F) ? F.toFixed(0) : '—';

  }

  calc();

  if('serviceWorker' in navigator){

    navigator.serviceWorker.register('./sw.js');

  }
